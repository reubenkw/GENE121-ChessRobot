char chessboard[][];

for(int i = 0; i < 8; i++)
{
	chessboard[i][0] = 'W';
}

for(int x = 0; x < 8; x++) {
	chessboard[x][1] = 'W';
}	

for(int y = 2; y < 6; y++) {
	for(int x = 0; x < 8; x++) {
		chessboard[x][y] = '.';
	}
}

for(int x = 0; x < 8; x++) {
	chessboard[x][6] = 'p';
}		
for(int i = 0; i < 8; i++)
{
	chessboard[i][6] = 'B';
}
for(int i = 0; i < 8; i++)
{
	chessboard[i][7] = 'B';
}
