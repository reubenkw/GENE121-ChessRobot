void waitForCPPFile(TFileHandle&fin, int desiredFirstElement) {
	int inNum = desiredFirstElement + 1;
	do{
		wait1Msec(2000);
		readIntPC(TFileHandle,inNum);
		//close file
	} while(inNum != desiredFirstElement);
}
